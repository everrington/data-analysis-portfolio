CRM Lead Generation

BigQuery SQL CRM RFM Segmentation Line OA Customer Analytics

Designed and implemented a production BigQuery SQL view to power CRM broadcast targeting for Line OA campaigns. The query builds an RFM-based segmentation pipeline — filtering reachable members, aggregating transaction history, scoring customers by recency and purchase frequency, and splitting the final audience into a broadcast lead group and a holdout control group for campaign measurement.

The output feeds directly into the campaign broadcast system, ensuring each message is sent only to customers with genuine repurchase potential — reducing broadcast waste and improving response rates.

-- ============================================================
--  CRM LEAD GENERATION — LINE BROADCAST TARGETING
--  Goal : Identify and segment customers for Line OA broadcast
--         based on purchase recency, frequency, and basket size
-- ============================================================

-- ── VARIABLES ─────────────────────────────────────────────
WITH _var AS (
  SELECT
    30                           AS lookback_days,
    CURRENT_DATE("Asia/Bangkok") AS end_date
),

-- ── PRODUCT FILTER ────────────────────────────────────────
_product AS (
  SELECT DISTINCT *
  FROM `[project].product_master`
  WHERE category_name IN ("SHAMPOO", "CONDITIONER")
    AND   brand_name     = "[Target Brand]"
),

-- ── REACHABLE MEMBERS ─────────────────────────────────────
_customer AS (
  SELECT *
  FROM `[project].member_master`
  WHERE is_line_oa_reachable IS TRUE
),

-- ── BRANCH FILTER ─────────────────────────────────────────
_branch AS (
  SELECT *
  FROM `[project].branch_master`
  WHERE storetype NOT IN ("[Excluded Format]")
),

-- ── TRANSACTION AGGREGATION ───────────────────────────────
_receipt AS (
  SELECT
    r.customer_code,
    c.member_no,
    r.doc_date,
    p.brand_name,
    SUM(r.sales_exc_vat + r.sales_vat) AS basket_spend

  FROM `[project].transactions` r
  INNER JOIN _product  p ON r.barcode      = p.barcode
  INNER JOIN _branch   b ON r.branch_code  = b.branch_code
  INNER JOIN _customer c ON r.customer_code = c.customer_code

  WHERE r.doc_date BETWEEN
          (SELECT end_date - lookback_days FROM _var)
          AND (SELECT end_date FROM _var)
    AND r.sales_qty     <> 0
    AND r.sales_exc_vat <> 0

  GROUP BY ALL
),

-- ── RFM BASE ──────────────────────────────────────────────
_rfm_base AS (
  SELECT
    customer_code,
    member_no,
    MIN(doc_date)            AS first_purchase_date,
    MAX(doc_date)            AS last_purchase_date,
    COUNT(DISTINCT doc_date) AS frequency,
    AVG(basket_spend)        AS avg_basket,
    MAX(basket_spend)        AS max_basket
  FROM _receipt
  GROUP BY ALL
),

-- ── INTERVAL + RECENCY ────────────────────────────────────
_rfm_scored AS (
  SELECT
    *,
    SAFE_DIVIDE(
      DATE_DIFF(last_purchase_date, first_purchase_date, DAY),
      frequency - 1
    )                                              AS avg_purchase_interval,
    DATE_DIFF(
      (SELECT end_date FROM _var), last_purchase_date, DAY
    )                                              AS recency_days
  FROM _rfm_base
),

-- ── BASKET BANDING + PERCENTILES ──────────────────────────
_rfm_banded AS (
  SELECT
    *,
    CASE
      WHEN avg_basket <=  50 THEN "฿1–50"
      WHEN avg_basket <= 100 THEN "฿51–100"
      WHEN avg_basket <= 150 THEN "฿101–150"
      WHEN avg_basket <= 200 THEN "฿151–200"
      WHEN avg_basket <= 250 THEN "฿201–250"
      WHEN avg_basket <= 300 THEN "฿251–300"
      WHEN avg_basket >  300 THEN "฿300+"
      ELSE "—"
    END AS avg_basket_range,
    CASE
      WHEN max_basket <=  50 THEN "฿1–50"
      WHEN max_basket <= 100 THEN "฿51–100"
      WHEN max_basket <= 200 THEN "฿151–200"
      WHEN max_basket >  300 THEN "฿300+"
      ELSE "—"
    END AS max_basket_range,
    PERCENTILE_CONT(avg_purchase_interval, 0.2 IGNORE NULLS) OVER () AS pct_20,
    PERCENTILE_CONT(avg_purchase_interval, 0.5 IGNORE NULLS) OVER () AS pct_50,
    PERCENTILE_CONT(avg_purchase_interval, 0.8 IGNORE NULLS) OVER () AS pct_80
  FROM _rfm_scored
),

-- ── FREQUENCY LABEL ───────────────────────────────────────
_rfm_freq AS (
  SELECT
    *,
    CASE
      WHEN frequency = 1                          THEN "one-time"
      WHEN avg_purchase_interval <= pct_20        THEN "q1 — frequent"
      WHEN avg_purchase_interval <= pct_50        THEN "q2 — regular"
      WHEN avg_purchase_interval <= pct_80        THEN "q3 — occasional"
      WHEN avg_purchase_interval >  pct_80        THEN "q4 — infrequent"
      ELSE "—"
    END AS frequency_segment
  FROM _rfm_banded
),

-- ── RECENCY LEAD LABEL ────────────────────────────────────
_recency_labeled AS (
  SELECT
    *,
    CASE
      WHEN recency_days <  pct_20                 THEN "Hot — recently bought (p0–20)"
      WHEN recency_days BETWEEN pct_20 AND pct_80 THEN "Warm — hot lead (p20–80)"
      WHEN recency_days >  pct_80                 THEN "Cold — churn risk (p80–100)"
    END AS lead_warmth
  FROM _rfm_freq
),

-- ── CONTROL GROUP SPLIT ───────────────────────────────────
_segmented AS (
  SELECT
    * EXCEPT (lead_group),
    CASE
      WHEN lead_group = 20 THEN "control group"
      ELSE                      "broadcast lead"
    END AS lead_group
  FROM (
    SELECT
      *,
      NTILE(20) OVER (
        PARTITION BY lead_warmth
        ORDER BY     member_no
      ) AS lead_group
    FROM _recency_labeled
    WHERE lead_warmth != "Hot — recently bought (p0–20)"
      AND max_basket   > 200
    ORDER BY customer_code
  )
)

-- ── FINAL OUTPUT ──────────────────────────────────────────
SELECT
  lead_group,
  lead_warmth,
  max_basket_range,
  COUNT(DISTINCT customer_code) AS customer_count
FROM _segmented
GROUP BY ALL
ORDER BY lead_group, lead_warmth, max_basket_range
